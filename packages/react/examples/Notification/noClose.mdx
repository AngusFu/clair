function Index() {
  const showNotification = () =>
    notification.info({
      title: "Notification without close button",
      content: "Some content...",
      showClose: false
    });
  return (
    <div className="s-wrapper--button">
      <Button type="primary" onClick={showNotification}>
        弹出无关闭按钮的通知
      </Button>
    </div>
  );
}
