## 自定义尺寸

自定义尺寸，适应在各种容器中展示。

```tsx live=true
function Demo() {
  let [selected, setSelectedVal] = React.useState("normal");

  const setSelected = e => {
    setSelectedVal(e.target.value);
  };

  return (
    <div>
      <RadioGroup value={selected} onChange={setSelected}>
        <Radio value="large">大</Radio>
        <Radio value="normal">正常</Radio>
        <Radio value="small">小</Radio>
      </RadioGroup>
      <br />
      <br />
      <h5 style={{ fontSize: "14px", fontWeight: 600, marginBottom: "20px" }}>
        产品信息
      </h5>
      <Descriptions bordered size={selected} column={2}>
        <DescriptionsItem label="产品名称">Clair Design</DescriptionsItem>
        <DescriptionsItem label="版本号">V1.0.0</DescriptionsItem>
        <DescriptionsItem label="开发团队">前端部</DescriptionsItem>
        <DescriptionsItem label="发布时间">
          2019-12-16 10:40:35
        </DescriptionsItem>
        <DescriptionsItem label="注册信息">设计公司</DescriptionsItem>
        <DescriptionsItem label="设计团队">用户体验部</DescriptionsItem>
        <DescriptionsItem label="备注信息">
          1.迭代持续中；
          <br />
          2.更多功能敬请期待；
          <br />
          ...
        </DescriptionsItem>
      </Descriptions>
      <br />
      <br />
      <h5 style={{ fontSize: "14px", fontWeight: 600, marginBottom: "20px" }}>
        产品信息
      </h5>
      <Descriptions size={selected}>
        <DescriptionsItem label="产品名称：">Clair Design</DescriptionsItem>

        </DescriptionsItem>

      </Descriptions>
    </div>
  );
}
```
