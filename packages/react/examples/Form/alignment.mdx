## 对齐方式

根据具体目标和制约因素，选择最佳的标签对齐方式。

```tsx live=true className=s-form-example
function Index() {
  const options = [
    { label: "右对齐", value: "right" },
    { label: "左对齐", value: "left" },
    { label: "上对齐", value: "top" }
  ];
  const [position, updateSelected] = React.useState(options[0].value);

  const onSelectedChange = e => {
    updateSelected(e.target.value);
  };

  return (
    <>
      <RadioGroup
        value={position}
        onChange={onSelectedChange}
        className="radio-group"
        style={{ marginBottom: 24 }}
      >
        {options.map(({ label, value }) => (
          <Radio value={value} key={label}>
            {label}
          </Radio>
        ))}
      </RadioGroup>
      <Form labelPosition={position} labelWidth={80}>
        <FormItem
          label="名称"
          name="name"
          as="input"
          className="c-input c-input--normal"
          placeholder="请输入名称"
        />
        <FormItem
          label="联系电话"
          name="phone"
          as="input"
          className="c-input c-input--normal"
          placeholder="请输入电话"
        />
        <FormItem
          label="联系人"
          name="contact"
          as="input"
          className="c-input c-input--normal"
          placeholder="请输入联系人"
        />
      </Form>
    </>
  );
}
```
