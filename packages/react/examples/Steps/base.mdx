# 基础用法

```tsx live=true
<Steps>
  <Step stepKey="1" title="第一步" />
  <Step stepKey="2" title="第二步" />
  <Step stepKey="3" title="第三步" />
  <Step stepKey="4" title="完成" />
</Steps>
```

### 设置 status

```tsx live=true
<Steps>
  <Step stepKey="1" title="第一步" status="finish" />
  <Step stepKey="2" title="第二步" status="error" />
  <Step stepKey="3" title="第三步" />
  <Step stepKey="4" title="完成" />
</Steps>
```

```tsx live=true
<Steps activeKey="3">
  <Step stepKey="1" title="第一步" />
  <Step stepKey="2" title="第二步" />
  <Step stepKey="3" title="第三步" />
  <Step stepKey="4" title="完成" />
</Steps>
```

```tsx live=true
<Steps>
  <Step stepKey="1" title="第一步" status="finish" description="第一步的描述" />
  <Step
    stepKey="2"
    title="第二步"
    status="process"
    description="第二步的描述"
  />
  <Step stepKey="3" title="第三步" description="第三步的描述" />
</Steps>
```
