  const showModal = () => setVisibility(true);

  return (
    <React.Fragment>

        打开弹窗

      <Modal
        title="Light Modal"
        visible={visible}

        light={true}
        onCancel={cancel}
        onConfirm={confirm}
      >
        <p>Some contents...</p>
        <p>Some contents...</p>
      </Modal>
    </React.Fragment>
  );
}
