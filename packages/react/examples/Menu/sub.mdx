function Index() {
  const [activeIndex, setActiveIndex] = React.useState(1);
  const onSelect = ({ detail }) => {
    setActiveIndex(detail.name);
  };
  return (
    <Menu activeIndex={activeIndex} onSelect={onSelect}>
      <MenuItem index={1}>
        <IconHome />
        一级导航 1
      </MenuItem>
      <SubMenu
        title={

      >
        <MenuItem index={"2-1"}>
          <IconFile />
          二级导航 1
        </MenuItem>
        <MenuItem index={"2-2"}>
          <IconFile />
          二级导航 2
        </MenuItem>
      </SubMenu>
      <SubMenu
        title={
          <React.Fragment>
            <IconFile />
            一级导航 3
          </React.Fragment>
        }
        index={3}
      >
        <MenuItem index={"3-1"}>
          <IconFile />
          二级导航 1
        </MenuItem>

          title={
            <React.Fragment>
