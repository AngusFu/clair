```tsx live=true
function Basic() {
  const getColumns = hidden => [
    { title: "来源类型", prop: "type", hidden },
    { title: "浏览量", prop: "pv" },
    { title: "访客数", prop: "uv" },
    { title: "IP 数", prop: "ip" }
  ];

  const [columns, setColumns] = React.useState(getColumns());

  const dataSource = [
    { type: "直接访问", pv: 13, uv: 9, ip: 8 },
    { type: "搜索引擎", pv: 11, uv: 7, ip: 6 },
    { type: "外部网站", pv: 32, uv: 27, ip: 22 }
  ];

  const onToggle = ({ target }) => {
    const { value } = target;
    setColumns(getColumns(value));
  };

  return (
    <div>
      <Checkbox onChange={onToggle}>来源类型</Checkbox>
      <Table columns={columns} dataSource={dataSource} rowKey="type" />
    </div>
  );
}
```
