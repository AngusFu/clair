## 受控下拉菜单

通过设置`visible`属性使`Dropdown`成为受控组件

```tsx live=true deps=message
function Index() {
  const [visible, setVisibility] = React.useState(false);

  const handleClick = e => {
    const {
      detail: { visible }
    } = e;
    setVisibility(visible);
  };

  const overlay = (
    <Dropdown.Menu>
      <Dropdown.Item>北京</Dropdown.Item>
      <Dropdown.Item disabled>上海</Dropdown.Item>
      <Dropdown.Item>广州</Dropdown.Item>
    </Dropdown.Menu>
  );

  return (
    <>
      <div>
        <Button type="primary" onClick={() => setVisibility(!visible)}>
          {visible ? "隐藏下拉" : "显示下拉"}
        </Button>
        &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;
        <Dropdown
          visible={visible}
          overlay={overlay}
          onVisibilityChange={handleClick}
        >
          <a>下拉菜单</a>
          &nbsp;
          <IconArrowDown />
        </Dropdown>
      </div>
    </>
  );
}
```
